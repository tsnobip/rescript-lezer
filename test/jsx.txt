# Simple JSX

let el = <div>Hello</div>

==>

Script(LetBinding(let,LetValueBinding(VariableDefinition,"=",JSXElement(JSXOpenTag(JSXStartTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag),JSXText,JSXCloseTag(JSXStartCloseTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag)))))

# Nested JSX with expression

let nested = <div><span>Hi</span>{expr}</div>

==>

Script(LetBinding(let,LetValueBinding(VariableDefinition,"=",JSXElement(JSXOpenTag(JSXStartTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag),JSXElement(JSXOpenTag(JSXStartTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag),JSXText,JSXCloseTag(JSXStartCloseTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag)),JSXExpressionContainer("{",VariableName,"}"),JSXCloseTag(JSXStartCloseTag,JSXIntrinsicElementName(JSXIdentifier),JSXEndTag)))))

# Self-closing JSX

let selfClosing = <img src="url" />

==>

Script(LetBinding(let,LetValueBinding(VariableDefinition,"=",JSXElement(JSXSelfClosingTag(JSXStartTag,JSXIntrinsicElementName(JSXIdentifier),JSXAttribute(JSXIdentifier,"=",String),JSXSelfCloseEndTag)))))

# Uppercase JSX component

let comp = <MyComponent>Hello</MyComponent>

==>

Script(LetBinding(let,LetValueBinding(VariableDefinition,"=",JSXElement(JSXOpenTag(JSXStartTag,JSXCustomComponentName(ModulePath(ModuleName)),JSXEndTag),JSXText,JSXCloseTag(JSXStartCloseTag,JSXCustomComponentName(ModulePath(ModuleName)),JSXEndTag)))))

# Uppercase JSX member expression

let frag = <React.Fragment><span /></React.Fragment>

==>

Script(LetBinding(let,LetValueBinding(VariableDefinition,"=",JSXElement(JSXOpenTag(JSXStartTag,JSXCustomComponentName(ModulePath(ModulePrefix,ModuleName)),JSXEndTag),JSXElement(JSXSelfClosingTag(JSXStartTag,JSXIntrinsicElementName(JSXIdentifier),JSXSelfCloseEndTag)),JSXCloseTag(JSXStartCloseTag,JSXCustomComponentName(ModulePath(ModulePrefix,ModuleName)),JSXEndTag)))))
